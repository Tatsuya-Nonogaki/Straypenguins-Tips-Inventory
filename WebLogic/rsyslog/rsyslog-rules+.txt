# WebLogic logs
if $programname == 'startWebLogic.sh' or $programname == 'stopWebLogic.sh' then {
    action(type="omfile"
           fileOwner="oracle" fileGroup="oracle" FileCreateMode="0600"
           iOBufferSize="8k"
           File="/var/log/weblogic/startWebLogic.log"
    )
    stop
}

           File="/var/log/weblogic/startManagedWebLogic.log"
    )
    stop
}
